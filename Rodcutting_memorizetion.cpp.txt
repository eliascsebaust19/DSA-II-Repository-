#include<bits/stdc++.h>
using namespace std;

vector<int> price;
int n;

// Recursive function
int RC(int i, int len){

    // Base case: no more items or no length left
    if(i > n || len == 0) return 0;

    // Skip case
    int skip = RC(i + 1, len);

    // Take case (if possible)
    int take = 0;
    if(i <= len){
        take = price[i] + RC(i, len - i);
    }

    // Return best of both
    return max(take, skip);
}

int main(){
    int len;
    cin >> n >> len;

    price.resize(n + 1);

    for(int i = 0; i <= n; i++){
        cin >> price[i];
    }

    cout << RC(1, len) << endl;

    return 0;
}
